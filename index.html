Code example

You have to write a function that will validate that a given binary tree is a binary search tree. The sort order is not predefined so it should work with either.

(Pinned from Codewars.com)

                      class Node {
    constructor(value, left = null, right = null){
      this.value = value
      this.left = left
      this.right = right
    } 
  }

  const isBST = node => {
    const arrayBST = doLeft(node)
    if(arrayBST[1] > arrayBST[0]) {
      for(let i = 0; i < arrayBST.length - 1; i++) {
        if (arrayBST[i] > arrayBST[i+1]) {
          return false
        }
      }
    } else {
      for(let i = 0; i < arrayBST.length - 1; i++) {
        if (arrayBST[i] < arrayBST[i+1]) {
          return false
        }
      }
    }
    return true
  };

  function doLeft(node, cb = []) {
    if(node) {
      doLeft(node.left, cb)
      cb.push(node.value)
      doLeft(node.right, cb)
    }
    return cb
  }
